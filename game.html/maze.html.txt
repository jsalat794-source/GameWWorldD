    <center>
        <h1>Случайный Лабиринт</h1>
        <p>Сложность: <b>СРЕДНЯЯ</b>. Доберись синим квадратом до зеленого финиша!</p>

        <!-- Игровое поле -->
        <canvas id="mazeCanvas" width="390" height="390" style="border: 4px solid white; background: #111;"></canvas>
        
        <br><br>
        
        <!-- Кнопка генерации нового лабиринта -->
        <button onclick="setupMaze()" style="padding: 10px 20px; border-radius: 10px; cursor: pointer; font-size: 16px; margin-right: 10px;">
            🔄 Другой лабиринт
        </button>

        <!-- Работающая кнопка возврата в меню -->
        <a href="index.html">

    <script>
        const canvas = document.getElementById("mazeCanvas");
        const ctx = canvas.getContext("2d");

        // Настройки для СРЕДНЕЙ сложности (13 клеток оптимальны по балансу)
        const cols = 13;
        const rows = 13;
        const size = 30; // Размер одной клетки увеличен до 30px, чтобы поле оставалось крупным

        let grid = [];
        let player = { x: 0, y: 0 };
        let goal = { x: cols - 1, y: rows - 1 };

        function Cell(r, c) {
            this.r = r;
            this.c = c;
            this.walls = [true, true, true, true]; // Верх, Право, Низ, Лев
            this.visited = false;
        }

        // Генерация лабиринта (каждый раз строит уникальный путь)
        function generateMaze() {
            grid = [];
            for (let r = 0; r < rows; r++) {
                grid[r] = [];
                for (let c = 0; c < cols; c++) {
                    grid[r][c] = new Cell(r, c);
                }
            }

            let stack = [];
            let current = grid[0][0];
            current.visited = true;

            while (true) {
                let neighbors = [];
                let r = current.r;
                let c = current.c;

                if (r > 0 && !grid[r - 1][c].visited) neighbors.push(grid[r - 1][c]);
                if (c < cols - 1 && !grid[r][c + 1].visited) neighbors.push(grid[r][c + 1]);
                if (r < rows - 1 && !grid[r + 1][c].visited) neighbors.push(grid[r + 1][c]);
                if (c > 0 && !grid[r][c - 1].visited) neighbors.push(grid[r][c - 1]);

                if (neighbors.length > 0) {
                    let next = neighbors[Math.floor(Math.random() * neighbors.length)];
                    next.visited = true;
                    stack.push(current);

                    // Ломаем стены между клетками
                    if (current.r === next.r + 1) { current.walls[0] = false; next.walls[2] = false; }
                    if (current.c === next.c - 1) { current.walls[1] = false; next.walls[3] = false; }
                    if (current.r === next.r - 1) { current.walls[2] = false; next.walls[0] = false; }
                    if (current.c === next.c + 1) { current.walls[3] = false; next.walls[1] = false; }

                    current = next;
                } else if (stack.length > 0) {
                    current = stack.pop();
                } else {
                    break;
                }
            }
        }

        // Рисование
        function draw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = "white";
            ctx.lineWidth = 3; // Стены стали толще и заметнее

            for (let r = 0; r < rows; r++) {
                for (let c = 0; c < cols; c++) {
                    let x = c * size;
                    let y = r * size;
                    let cell = grid[r][c];

                    if (cell.walls[0]) { ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x + size, y); ctx.stroke(); }
                    if (cell.walls[1]) { ctx.beginPath(); ctx.moveTo(x + size, y); ctx.lineTo(x + size, y + size); ctx.stroke(); }
                    if (cell.walls[2]) { ctx.beginPath(); ctx.moveTo(x + size, y + size); ctx.lineTo(x, y + size); ctx.stroke(); }
                    if (cell.walls[3]) { ctx.beginPath(); ctx.moveTo(x, y + size); ctx.lineTo(x, y); ctx.stroke(); }
                }
            }

            // Рисуем финиш
            ctx.fillStyle = "#2ecc71";
            ctx.fillRect(goal.x * size + 6, goal.y * size + 6, size - 12, size - 12);

            // Рисуем игрока
            ctx.fillStyle = "#3498db";
            ctx.fillRect(player.x * size + 6, player.y * size + 6, size - 12, size - 12);
        }

        function setupMaze() {
            player.x = 0;
            player.y = 0;
            generateMaze();
            draw();
        }

        // Управление игроком
        document.addEventListener("keydown", function(e) {
            let currentCell = grid[player.y][player.x];
            let moved = false;

            if ((e.key === "ArrowUp" || e.keyCode === 38) && !currentCell.walls[0]) { player.y--; moved = true; }
            if ((e.key === "ArrowRight" || e.keyCode === 39) && !currentCell.walls[1]) { player.x++; moved = true; }
            if ((e.key === "ArrowDown" || e.keyCode === 40) && !currentCell.walls[2]) { player.y++; moved = true; }
            if ((e.key === "ArrowLeft" || e.keyCode === 37) && !currentCell.walls[3]) { player.x--; moved = true; }

            if (moved) {
                e.preventDefault();
                draw();
                checkWin();
            }
        });

        function checkWin() {
            if (player.x === goal.x && player.y === goal.y) {
                setTimeout(() => {
                    alert("Отлично! Лабиринт пройден.");
                    setupMaze(); // Автоматически загружает новое случайное поле
                }, 100);
            }
        }

        setupMaze();
    </script>
